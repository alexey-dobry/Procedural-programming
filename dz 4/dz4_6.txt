int romanleters(char romnumb)
{
    if (romnumb == 'I') {
        return 1;
    }
    else if (romnumb == 'V') {
        return 5;
    }
    else if (romnumb == 'X') {
        return 10;
    }
    else if (romnumb == 'L') {
        return 50;
    }
    else if (romnumb == 'C') {
        return 100;
    }
    else if (romnumb == 'D') {
        return 500;
    }
    else if (romnumb == 'M') {
        return 1000;
    }
}

void dz_6()
{
    string str;
    cout << "Введите римское число: ";
    cin >> str;
    int i;
    int ans = 0; 
    for (i = 0; i <= str.size(); i++)
        if ((i < str.size() - 1) && (romanleters(str[i]) < romanleters(str[i + 1]))) {
            ans -= romanleters(str[i]);
        }
        else {
            ans += romanleters(str[i]);
        }
    cout << ans;
}

int main()
{
    dz_6();
    return 0;
