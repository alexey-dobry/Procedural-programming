void dz3_2()
{
    float S, n, m0, r, eps, m;
    eps = 0.01;
    cout << "Введите займ S: ";
    cin >> S;
    cout << "Введите ежемесячный платёж m: ";
    cin >> m0;
    cout << "Введите количество лет n: ";
    cin >> n;
    int i;
    for (i = 1; i <= 100; i++) {
        r = i / 100.0;
        if ((12 * (pow(1 + r, n) - 1)) == 0) {
            cout << "Ошибка";
        }
        else {
            m = S * r * pow(1 + r, n) / (12 * (pow(1 + r, n) - 1));
        }
        if (fabs(m - m0) < eps) {
            cout << "p = " << i;
        }
    }
}
